# `函数`

```javascript
const courses = {
    // shaopeng
    '00': {
        '06': 'reviews'
    },
    // English
    '03': {
        '001': '高考英语 单词'
    },
    // Software
    '04': {
        '001': 'Obsidian',
        '002': 'Visual Studio Code',
        '003': 'IDEA'
    },
    // BigData
    '06': {
        '001': 'Linux'
    }
};

// 学科-课程-xx...
const match = title.match(/^\[m\](.+)  ([0-9]{2})-([0-9]{2,3})(.+)$/);
if (match) {
    const heading = match[1];
    const subject = match[2];
    const course = match[3];
    const rest = match[4];
    if (courses[subject] && courses[subject][course]) {
        return `${heading}\n${subject}-${courses[subject][course]}${rest}`;
    }
}
// 学科 || 学科-课程
if (title.match(/^\[m\].+$/)) {
    title = title.replace("[m]", "");
}
return title.replace("  ", "\n");
```
